### 3. 性能测试脚本

```python
import time
import asyncio
from src.core.agent import AntiFraudAgent, AgentInput

async def performance_test():
    agent = AntiFraudAgent(user_id="perf_test")
    
    # 响应时间测试
    start = time.time()
    input_data = AgentInput(text="测试文本", modality="text")
    result = await agent.process(input_data)
    duration = time.time() - start
    
    print(f"响应时间: {duration:.3f}秒")
    
    # 压力测试
    for i in range(100):
        await agent.process(input_data)
    
    print(f"100次请求平均响应时间: {(time.time() - start)/100:.3f}秒")
```
